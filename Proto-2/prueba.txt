

int main() {
    // 1. Inicializar la librería (Obligatorio)
    if (sodium_init() < 0) {return 1;}

    // --- DATOS DEL USUARIO ---
    const char *password = "password123"; // La clave del usuario
    const char *mensaje_original = "Este es un dato personal secreto";
    size_t mensaje_len = strlen(mensaje_original);

    // --- PREPARACIÓN DE ESPACIO ---
    // Usamos unsigned char porque son bytes puros
    unsigned char salt[crypto_pwhash_SALTBYTES];
    unsigned char nonce[crypto_secretbox_NONCEBYTES];
    unsigned char key[crypto_secretbox_KEYBYTES];
    
    // El texto cifrado necesita un poco más de espacio para el "Tag" de seguridad
    unsigned char cifrado[mensaje_len + crypto_secretbox_MACBYTES];
    char descifrado[mensaje_len + 1];

    // --- PROCESO DE CIFRADO ---

    // A. Generar valores aleatorios únicos
    randombytes_buf(salt, sizeof salt);
    randombytes_buf(nonce, sizeof nonce);

    // B. Convertir el password en una llave real
    crypto_pwhash(key, sizeof key, password, strlen(password), salt,
                  crypto_pwhash_OPSLIMIT_INTERACTIVE, 
                  crypto_pwhash_MEMLIMIT_INTERACTIVE, crypto_pwhash_ALG_DEFAULT);

    // C. Cifrar el mensaje
    crypto_secretbox_easy(cifrado, (const unsigned char *)mensaje_original, 
                          mensaje_len, nonce, key);

    // Mostramos el mensaje cifrado.2
    printf("Mensaje cifrado (hex): ");
    for(int i = 0; i < sizeof cifrado; i++) {printf("%02x", cifrado[i]);};
    printf("\n");

    // --- PROCESO DE DESCIFRADO ---

    // (En la vida real, sacarías la llave de la misma forma usando el password y el salt)
    if (crypto_secretbox_open_easy((unsigned char *)descifrado, cifrado, sizeof cifrado, nonce, key) != 0) {
        printf("Error: Contraseña incorrecta o datos manipulados.\n");
    } else {
        descifrado[mensaje_len] = '\0'; // Cerrar el string
        printf("Mensaje recuperado: %s\n", descifrado);
    }

    // --- LIMPIEZA ---
    sodium_memzero(key, sizeof key); // Borrar la llave de la RAM

    return 0;
}
